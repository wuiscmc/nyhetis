post '/session' do 
  begin
    content_type :json
    session_id = @authentication_controller.create_session(params)
    {session: session_id}.to_json
  rescue AuthenticationFailedError
    halt 401, headers, {code: 401, message: 'Unauthorized'}.to_json
  end
end

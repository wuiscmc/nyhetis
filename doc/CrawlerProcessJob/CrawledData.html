def initialize(params = {})
	params.each do |name, value|
     value = case value 
     when 'http://localhost:7541/'
       'diariojaen.es'
     when 'http://localhost:7542/'
       'ideal.es'
     when 'http://localhost:7543/'
       'andaluciainformacion.es'
     else
       value
     end

		send("#{name}=", value)
	end
end

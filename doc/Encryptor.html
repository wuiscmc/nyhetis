def self.decrypt(string)
  decipher = encryptor
  decipher.decrypt

  decrypted = Base64.decode64(string)

def self.encrypt(string)
  cipher = encryptor
  cipher.encrypt
  cipher.key = APPLICATION_KEY
  encrypted = cipher.update(string) + cipher.final
  Base64.encode64(encrypted).to_json
end
